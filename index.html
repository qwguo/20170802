$(function() {
  var span1 = $('#span1'),
    span2 = $('#span2'),
    times = $('#time');
  countDown({
    st: '2018-01-03 14:59:30',	//开始时间
    et: '2018-01-03 14:49:35',	//结束时间
    nt: '2018-01-03 14:49:28',	//现在时间用于读取服务器时间
    o: times,			//jquery的dom元素
    sfun: function() {		//开始前要执行的方法
      span1.html('还有：');
      span2.html('开始！');
    },
    nfun: function() {		//已经开始要执行的方法
      span1.html('还有：');
      span2.html('结束！');
    },
    efun: function() {		//已经结束要执行的方法
      span1.html('活动：');
      span2.html('结束！');
      times.hide();
    }
  });
});
